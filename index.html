Electric Tentacles 


class Tentacle {
    constructor(x, y, l, n, a) {
        this.pos = { x, y };
        this.length = l;
        this.segments = n;
        this.angle = a;
        this.branches = [];
        
        // Генерация сегментов
        for (let i = 0; i < n; i++) {
            this.branches.push({
                x: x + Math.cos(a) * l * i,
                y: y + Math.sin(a) * l * i,
                width: l * (1 - i/n)
            });
        }
    }

    update() {
        // Анимация щупальца...
    }
}
